Macos:
git clone -b openssl-3.5.0 --recursive https://github.com/openssl/openssl openssl-arm
git clone -b openssl-3.5.0 --recursive https://github.com/openssl/openssl openssl-x64
cd openssl-arm
./Configure zlib darwin64-arm64-cc no-apps no-docs no-filenames no-shared no-weak-ssl-ciphers --release
make -s
sudo make install -s
cd ../openssl-x64
./Configure zlib darwin64-x86_64-cc no-apps no-docs no-filenames no-shared no-weak-ssl-ciphers --release
make -s
cd ..
mkdir openssl-fat
lipo -create openssl-arm/libcrypto.a openssl-x64/libcrypto.a -output openssl-fat/libcrypto.a
lipo -create openssl-arm/libssl.a openssl-x64/libssl.a -output openssl-fat/libssl.a
    sudo cp openssl-fat/*.a /usr/local/lib
    cd back to vcpkg dir
